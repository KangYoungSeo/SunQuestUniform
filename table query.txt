/*데이터베이스 생성(초기 설정)*/

CREATE DATABASE sunquest;



/*데이터베이스 사용*/
use sunquest;


/* USER : 고객(사용자) 정보*/

/*생성 쿼리*/

create table USER(
CUS_NUM INT NOT NULL,
ID VARCHAR(12) NOT NULL,
PW VARCHAR(12) NOT NULL,
EMAIL VARCHAR(30) NOT NULL,
PRIMARY KEY(CUS_NUM));


/*PRODUCT : 자수정보*/

/*생성 쿼리*/

create table PRODUCT(
P_ID INT NOT NULL,
P_CAT VARCHAR(12),
LCH_DATE DATE,
P_NAME VARCHAR(30),
P_DETAIL VARCHAR(200),
FABRIC VARCHAR(200),
CARE_INS VARCHAR(200),
PRICE INT,
SORT VARCHAR(7),
SIZE VARCHAR(4),
SEX VARCHAR(1),
COLOR VARCHAR(8),
EBM_PLACE VARCHAR(5),
PRIMARY KEY(P_ID));








/*EMBROID : 자수정보*/

/*생성 쿼리*/

create table EMBROID(
CUS_NUM INT,
P_ID INT,
EMB_PLACE VARCHAR(5),
FONT VARCHAR(12),
SIZE INT,
COLOR VARCHAR(8),
DETAIL VARCHAR(10),
PRIMARY KEY(CUS_NUM, P_ID));









/*BASKET : 장바구니*/

/*생성 쿼리*/

create table BASKET(
CUS_NUM INT,
P_ID INT,
QUANT VARCHAR(5),
SIZE VARCHAR(4),
SORT VARCHAR(7),
COLOR VARCHAR(8),
EBM_STAT BOOLEAN,
ORDER_STAT BOOLEAN,
PRIMARY KEY(CUS_NUM, P_ID));
